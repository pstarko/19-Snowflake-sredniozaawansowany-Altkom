--Jeśli chcę to zautomatyzować to TASK (pamiętaj, że tam może być TYLKO JEDNO POLECENIE DML
SELECT
  *
FROM
  EMP2

truncate table emp2;

CREATE OR REPLACE TASK load_csv_from_stage_task
    WAREHOUSE = COMPUTE_WH
    SCHEDULE = '1 MINUTE'
AS
EXECUTE IMMEDIATE $$
BEGIN    
    COPY INTO emp2
    FROM @MY_STAGE
    FILE_FORMAT = (TYPE = 'CSV' SKIP_HEADER = 1
    FIELD_DELIMITER = ';')
    FORCE = TRUE
    ON_ERROR = 'CONTINUE';
END;
$$;


ALTER TASK load_csv_from_stage_task RESUME;  
EXECUTE TASK load_csv_from_stage_task;


SELECT * FROM TABLE(INFORMATION_SCHEMA.TASK_HISTORY(TASK_NAME=>'LOAD_CSV_FROM_STAGE_TASK')) ORDER BY SCHEDULED_TIME DESC LIMIT 5;
