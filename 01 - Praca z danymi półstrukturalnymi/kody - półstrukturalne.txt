--A1) Rozplątanie tablicy items do wierszy
SELECT
  o.order_id,
  o.customer,
  f.value:sku::string AS sku,
  f.value:qty::int    AS qty
FROM orders o, LATERAL FLATTEN(input => o.items) f
ORDER BY o.order_id, sku;

/*
f.value to element tablicy (JSON object).
Rzutujemy typy: ::string, ::int. 
*/

--Ćwiczenie: Policz liczbę spłaszczonych wierszy versus oryginalna liczba zamówień.

SELECT
  COUNT(*) AS flattened_rows,
  (SELECT COUNT(*) FROM orders) AS orders_rows;

/*
Część B: Wielopoziomowe zagnieżdżenia (np. options)
Kolumna items zawiera czasem pole options jako tablicę. Spłaszczymy najpierw items, następnie options
*/

SELECT
  o.order_id,
  f1.value:sku::string AS sku,
  f1.value:qty::int    AS qty,
  f2.value::string     AS option_value
FROM orders o,
    LATERAL FLATTEN(input => o.items) f1,
    LATERAL FLATTEN(input => f1.value:options) f2
ORDER BY o.order_id, sku, option_value;

--outer - obsługa pustych
SELECT
  o.order_id,
  f1.value:sku::string AS sku,
  f1.value:qty::int    AS qty,
  f2.value::string     AS option_value
FROM orders o,
    LATERAL FLATTEN(input => o.items) f1,
    LATERAL FLATTEN(input => f1.value:options, outer => TRUE) f2
ORDER BY o.order_id, sku, option_value;

--Agregacje - ilość sztuk per SKU
WITH flat AS (
  SELECT
    f.value:sku::string AS sku,
    f.value:qty::int    AS qty
  FROM orders o,
  LATERAL FLATTEN(input => o.items) f
)
SELECT
  sku,
  SUM(qty) AS total_qty
FROM flat
GROUP BY 1
ORDER BY total_qty DESC;

/*
OBJECT_KEYS w Snowflake (JSON / VARIANT)
OBJECT_KEYS() to funkcja Snowflake, która wyciąga listę kluczy z obiektu JSON (czyli z wartości typu OBJECT trzymanej zwykle w kolumnie VARIANT).

Składnia
    OBJECT_KEYS(<variant_expr>)

Wejście: wyrażenie typu VARIANT (najczęściej wynik PARSE_JSON(...) albo kolumna VARIANT)
Działa sensownie tylko, gdy VARIANT jest obiektem JSON (np. { "a": 1, "b": 2 })
Wyjście: ARRAY kluczy (jako wartości VARIANT w tablicy)

*/
--przykład podstawowy
    SELECT OBJECT_KEYS(PARSE_JSON('{"a":1,"b":2,"nested":{"x":10}}')) AS keys;

    SELECT OBJECT_KEYS('{"a":1,"b":2,"nested":{"x":10}}') AS keys; --Invalid argument types for function 'OBJECT_KEYS': (VARCHAR(31))

/*
Gdy masz JSON jako tekst
Jeśli kolumna jest VARCHAR z JSON-em, najpierw parsujesz:
*/
     SELECT OBJECT_KEYS(PARSE_JSON(json_text_col)) AS keys
        FROM my_table;

/*
Iteracja po kluczach (częsty wzorzec: FLATTEN)
Żeby dostać “po jednym kluczu na wiersz”:
*/
    SELECT
      f.value::string AS key_name
    FROM TABLE(FLATTEN(
                  INPUT => OBJECT_KEYS(PARSE_JSON('{"a":1,"b":2}'))
                )) f;


    
/*
Jak z kluczy przejść do wartości
Możesz połączyć OBJECT_KEYS + FLATTEN i odczytać wartości z obiektu:
*/

    WITH src AS (
      SELECT PARSE_JSON('{"a":1,"b":2}') AS obj
    )
    SELECT
      f.value::string AS key,
      src.obj[f.value::string]           AS val
    FROM src,
         LATERAL FLATTEN(INPUT => OBJECT_KEYS(src.obj)) f;
