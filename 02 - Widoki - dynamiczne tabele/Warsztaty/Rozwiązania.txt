--zadanie 1
CREATE OR REPLACE VIEW V_ORDERS_CLEAN AS
SELECT
    ORDER_ID,
    ORDER_DATE,
    CUSTOMER_ID,
    PRODUCT_ID,
    STORE_ID,
    PROMO_ID,
    QUANTITY,
    UNIT_PRICE,
    DISCOUNT_PCT,
    ORDER_STATUS,
    CHANNEL,
    REGION,
    ROUND(QUANTITY * UNIT_PRICE * (1 - DISCOUNT_PCT / 100), 2) AS NET_AMOUNT
FROM FACT_ORDERS_RAW
WHERE ORDER_STATUS <> 'cancelled'
order by ORDER_ID;

desc table V_ORDERS_CLEAN;

select * from V_ORDERS_CLEAN;

--zadanie 2
CREATE OR REPLACE MATERIALIZED VIEW MV_DAILY_SALES AS
SELECT
    ORDER_DATE,
    REGION,
    CHANNEL,
    COUNT(*) AS ORDERS_CNT,
    SUM(QUANTITY) AS UNITS_SOLD,
    ROUND(SUM(ROUND(QUANTITY * UNIT_PRICE * (1 - DISCOUNT_PCT / 100), 2)), 2) AS REVENUE
FROM FACT_ORDERS_RAW
GROUP BY ORDER_DATE, REGION, CHANNEL;

desc table MV_DAILY_SALES;
select * from MV_DAILY_SALES


--zadanie 3
CREATE OR REPLACE DYNAMIC TABLE DT_ORDERS_CURATED
TARGET_LAG = '15 minutes'
WAREHOUSE = COMPUTE_WH
REFRESH_MODE = INCREMENTAL
AS
SELECT
    o.ORDER_ID,
    o.ORDER_DATE,
    c.CUSTOMER_NAME,
    c.SEGMENT,
    p.PRODUCT_NAME,
    o.QUANTITY,
    ROUND(o.QUANTITY * o.UNIT_PRICE * (1 - o.DISCOUNT_PCT / 100), 2) AS NET_AMOUNT
FROM FACT_ORDERS_RAW o
JOIN DIM_CUSTOMERS c
    ON o.CUSTOMER_ID = c.CUSTOMER_ID
JOIN DIM_PRODUCTS p
    ON o.PRODUCT_ID = p.PRODUCT_ID
JOIN DIM_STORES s
    ON o.STORE_ID = s.STORE_ID
WHERE o.ORDER_STATUS = 'completed';

desc table DT_ORDERS_CURATED;
