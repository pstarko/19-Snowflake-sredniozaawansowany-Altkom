Snowflake Stream nie śledzi operacji SQL (INSERT/UPDATE/DELETE), tylko zmiany na poziomie wierszy w micro-partycjach.

UPDATE w Snowflake to wewnętrznie DELETE + INSERT:

stary wiersz → wpis z METADATA$ACTION = 'DELETE'
nowy wiersz → wpis z METADATA$ACTION = 'INSERT'

Dlaczego więc METADATA$ISUPDATE = FALSE?
Kolumna METADATA$ISUPDATE = TRUE pojawia się tylko wtedy, gdy w tym samym batch'u zmian stream widzi parę DELETE+INSERT dla tego samego klucza — co sygnalizuje logiczny UPDATE.

CREATE OR REPLACE TABLE orders_target LIKE orders;  -- pusta kopia struktury

--  MERGE, który:
-- - dla INSERT (i "nowej" części UPDATE) wstawia/aktualizuje wiersz,
-- - dla DELETE usuwa wiersz z docelowej.
-- to może generować bład przy UPDATE (duplicate row)
MERGE INTO orders_target t
USING (
  SELECT METADATA$ACTION AS action, id, status
  FROM orders_stream
) s
ON t.id = s.id
WHEN MATCHED AND s.action = 'DELETE' THEN DELETE
WHEN MATCHED AND s.action = 'INSERT' THEN UPDATE SET t.status = s.status
WHEN NOT MATCHED AND s.action = 'INSERT' THEN INSERT (id, status) VALUES (s.id, s.status);

MERGE INTO orders_target t
USING (
  WITH s AS (
    SELECT
      METADATA$ACTION AS action,
      METADATA$ISUPDATE AS is_update,
      METADATA$ROW_ID AS row_id,
      id,
      status
    FROM orders_stream
  ),
  -- Wybieramy najnowszy wiersz na klucz (jeśli był UPDATE, wygra INSERT z nowszym row_id)
  s2 AS (
    SELECT s.*
    FROM (
      SELECT
        s.*,
        ROW_NUMBER() OVER (PARTITION BY id ORDER BY row_id DESC) AS rn
      FROM s
    ) s
    WHERE s.rn = 1
  )
  SELECT
    -- jeśli najnowszy to INSERT -> traktuj jako INSERT/UPSERT,
    -- jeśli najnowszy to DELETE -> traktuj jako DELETE
    action,
    id,
    status
  FROM s2
) x
ON t.id = x.id
WHEN MATCHED AND x.action = 'DELETE' THEN DELETE
WHEN MATCHED AND x.action = 'INSERT' THEN UPDATE SET t.status = x.status
WHEN NOT MATCHED AND x.action = 'INSERT' THEN INSERT (id, status) VALUES (x.id, x.status);
