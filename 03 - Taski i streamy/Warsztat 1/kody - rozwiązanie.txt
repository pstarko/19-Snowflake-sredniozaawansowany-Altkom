CREATE OR REPLACE TABLE EMP (
  last_name VARCHAR,
  job_id VARCHAR,
  salary NUMBER
);

-- Tabela LOG_EMP – zapis logów operacji
CREATE OR REPLACE TABLE LOG_EMP (
  id INT AUTOINCREMENT,
  dataczas TIMESTAMP,
  iloscwierszy NUMBER,
  status VARCHAR
);

CREATE OR REPLACE FILE FORMAT my_csv_format
  TYPE = 'CSV'
  FIELD_DELIMITER = ';'
  SKIP_HEADER = 1
  FIELD_OPTIONALLY_ENCLOSED_BY = '"';



-- Utworzenie stage
CREATE OR REPLACE STAGE my_stage;

--Dane z pliku CSV zostaną wgrane do tabeli EMP przy użyciu polecenia COPY INTO:

COPY INTO EMP
FROM @my_stage/emp.csv
FILE_FORMAT = (FORMAT_NAME = 'my_csv_format') FORCE = TRUE; --wymuszam ponowne ładowanie inaczej zostaje w pamięci stary plik

select * from emp


CREATE OR REPLACE TASK task_log_emp
  WAREHOUSE = compute_wh
  SCHEDULE = 'USING CRON * * * * * Europe/Warsaw'  -- co minutę
AS
INSERT INTO LOG_EMP (dataczas, iloscwierszy, status)
SELECT CURRENT_TIMESTAMP, COUNT(*), 'Sprawdzone'
FROM EMP;
